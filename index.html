租約清單
每筆租約對應一個區域（如 4FA、4FB）。設定區域總坪數、公設坪數後，系統自動計算各員工分攤比例。

連線狀態
尚未設定 Google Sheets 連線。
設定 Google Sheets 連線
此功能需要建立 Google Apps Script Web App 作為中介層，設定一次後即可使用。
1
前往 Google Sheets，建立一個新的試算表，複製網址中的 Spreadsheet ID（位於 /d/ 和 /edit 之間）
2
下方點「🔑 產生 Token」，複製產生的 Token 備用
3
在試算表中點選 「擴充功能」→「Apps Script」，將下方程式碼貼入，並把第 1、2 行的 YOUR_SPREADSHEET_ID_HERE 和 YOUR_SECRET_TOKEN_HERE 換成你的實際值後儲存
4
在 Apps Script 中點選 「部署」→「新增部署」→「Web 應用程式」，執行身分選「我」，存取權選「任何人」，取得部署網址
5
將所有欄位填入下方並點「💾 儲存設定」，再點「🔗 測試連線」確認
Apps Script 程式碼（點擊複製）
複製 const SPREADSHEET_ID = 'YOUR_SPREADSHEET_ID_HERE'; const SECRET_TOKEN = 'YOUR_SECRET_TOKEN_HERE'; // ← 與系統設定的 Token 相同 function doPost(e) { try { const payload = JSON.parse(e.postData.contents); // ── Token 驗證：token 不符直接拒絕 ── if (payload.token !== SECRET_TOKEN) { return ContentService.createTextOutput(JSON.stringify({ok:false,msg:'Unauthorized'})) .setMimeType(ContentService.MimeType.JSON); } const { action, data, version } = payload; const ss = SpreadsheetApp.openById(SPREADSHEET_ID); if (action === 'save') { let sheet = ss.getSheetByName('data') || ss.insertSheet('data'); sheet.clearContents(); sheet.getRange(1,1).setValue(JSON.stringify(data)); let vh = ss.getSheetByName('versions') || ss.insertSheet('versions'); const ts = new Date().toISOString(); vh.appendRow([ts, version?.note || '自動儲存', JSON.stringify(data)]); return ContentService.createTextOutput(JSON.stringify({ok:true,ts})) .setMimeType(ContentService.MimeType.JSON); } if (action === 'load') { let sheet = ss.getSheetByName('data'); if (!sheet || sheet.getLastRow() < 1) return ContentService.createTextOutput(JSON.stringify({ok:true,data:null})).setMimeType(ContentService.MimeType.JSON); const raw = sheet.getRange(1,1).getValue(); return ContentService.createTextOutput(JSON.stringify({ok:true,data:JSON.parse(raw)})) .setMimeType(ContentService.MimeType.JSON); } if (action === 'history') { let vh = ss.getSheetByName('versions'); if (!vh) return ContentService.createTextOutput(JSON.stringify({ok:true,versions:[]})).setMimeType(ContentService.MimeType.JSON); const rows = vh.getDataRange().getValues().reverse().slice(0,50); const versions = rows.map(r => ({ts: r[0], note: r[1], data: r[2]})); return ContentService.createTextOutput(JSON.stringify({ok:true,versions})) .setMimeType(ContentService.MimeType.JSON); } return ContentService.createTextOutput(JSON.stringify({ok:false,msg:'unknown action'})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({ok:false,msg:err.toString()})) .setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { return ContentService.createTextOutput(JSON.stringify({ok:false,msg:'Use POST'})) .setMimeType(ContentService.MimeType.JSON); }
此 Token 需與 Apps Script 第 2 行的 SECRET_TOKEN 相同，只存在你的瀏覽器中，不會上傳到 GitHub。
